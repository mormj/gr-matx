########################################################################

    "pygccxml"
    "import pygccxml"
    PYGCCXML_FOUND
    )

########################################################################

list(APPEND matx_python_files
    python_bindings.cc)

   ../..
   gr::matx
   "${matx_python_files}")

install(TARGETS matx_python DESTINATION ${GR_PYTHON_DIR}/matx COMPONENT pythonapi)
